54 template <class T, int N>
55 class DynamicBag : NoCopy
56 {
57 public:
58 int size() const { return _size; }
59 void ensure(int new_capacity) {
60 if (new_capacity > _capacity) {
61 _capacity = std::max(_capacity*2, new_capacity);
62 _arr = static_cast<T*>(
63 realloc(_arr, sizeof(T)*_capacity));
64 }
65 }
66 void push(const T& val) {
67 if (_size == _capacity) {
68 _capacity *= 2;
69 _arr = static_cast<T*>(
70 realloc(_arr, sizeof(T)*_capacity));
71 }
72 _arr[_size] = val;
73 ++_size;
74 }
75 T pop() {
76 return _arr[--_size];
77 }
78 ~DynamicBag() {
79 free(_arr);
80 }
81
82 T& operator[](int idx) { return _arr[idx]; }
83 const T& operator[](int idx) const { return _arr[idx]; }
84 private:
85 T* _arr = static_cast<T*>(malloc(sizeof(T) * N));
86 int _size = 0;
87 int _capacity = N;
88 };

122 template <class T>
123 class PackedStorage final : public NoInstance
124 {
125 public:
126 static void add(const ent_type ent, const T& val) {
127 _idToComp.ensure(ent.id+1);
128 _idToComp[ent.id] = _comps.size();
129 _comps.push(val);
130 _compToId.push(ent.id);
131 }
132 static void del(const ent_type ent) {
133 int idx = _idToComp[ent.id];
134 const id_type last = _compToId.pop();
135
136 _comps[idx] = _comps.pop();
137 _compToId[idx] = last;
138 _idToComp[last] = idx;
139 }
140 static T& get(const ent_type ent) {
141 return _comps[_idToComp[ent.id]];
142 }
143 private:
144 static inline Bag<T,InitialPackedSize> _comps;
145 static inline Bag<int,InitialEntities> _idToComp;
146 static inline Bag<id_type,InitialPackedSize> _compToId;
147 __attribute__((used)) static inline Register<T> _reg{del};

154 _idToComp.ensure(ent.id+1);

157 _idToComp[ent.id] = idx;

214 class World final : public NoInstance
215 {
216 static inline Bag<Mask,InitialEntities> _masks;
217 static inline Bag<id_type,IdBagSize> _ids;
218 static inline id_type _maxId = -1;
219 static auto& _deleters() {
220 static Bag<DeleteFunc,MaxComponents> _deleters;
221 return _deleters;
222 }
223 public:
226 static ent_type createEntity() {
227 if (_ids.size() > 0)
228 return {_ids.pop()};
229 _masks.push(Mask{});
230 return {++_maxId};
231 }
232
234 static void deleteEntity(ent_type ent) {
235 if constexpr (CallbackOnDelete) {
236 Mask m = _masks[ent.id];
237 int ctz;
238 while ((ctz = m.ctz()) >= 0) {
239 if (_deleters()[ctz] != nullptr)
240 _deleters()[ctz](ent);
241 m.clear(Mask::bit(ctz));
242 }
243 }
244 _masks[ent.id].clear();
245 _ids.push(ent.id);
246 }
